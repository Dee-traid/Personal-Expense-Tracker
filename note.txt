			==== Database Entities ====
				- User account:
					id pk not null varchar
					user_name varchar(100) not null
					email varchar(100) not null
					phone_number varchar(100) not null
					password varchar(500) not null
					income float
					created_at Timestamp
					updated_at Timestamp

				- Category table:
					id pk varchar not null
					category_name not null varchar(225)
					description varchar(100)
					created_at Timestamp
					updated_at Timestamp

				- Expense:
					id pk not null 
					category_id referencing category table
					user_id referencing user table 
					expense_name(100) not null
					date Date
					description of expense
					amount float 
					created_at Time stamp
					updated_at TimeStamp

				- Budget: 
					id pk not null
					user_id referencing user table id
					category_id referencing category table id
					amount float
					budget_start_date Date
					budget_end_date Date
					created_at Timestamp
					updated_at Timpstamp

				- Reminder: 
					id pk not null
					user_id referencing users id
					expense_id referencing expense id
					reminder date Date
					reminder time Timestamp
					created at Timestamp
					updated at
